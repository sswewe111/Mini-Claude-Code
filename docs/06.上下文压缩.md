# 上下文压缩

***上下文不是越多越好，而是要把“仍然有用的部分”留在活跃工作面里。***

**怎样在不丢掉主线连续性的前提下，把活跃上下文重新腾出空间。**

上下文压缩的核心不是删掉历史，也不是单纯减少字数，而是把细节搬离活跃上下文，同时保住模型继续完成任务所需的连续性。

```python
第 1 层：大结果不直接塞进上下文
  -> 写到磁盘，只留预览

第 2 层：旧结果不一直原样保留
  -> 替换成简短占位

第 3 层：整体历史太长时
  -> 生成一份连续性摘要

tool output
   |
   +-- 太大 -----------------> 保存到磁盘 + 留预览
   |
   v
messages
   |
   +-- 太旧 -----------------> 替换成占位提示
   |
   v
if whole context still too large:
   |
   v
compact history -> summary
```

## 2、上下文压缩常量

```python
CONTEXT_LIMIT = 50000#当消息历史长度估算超过 50000 时，触发自动压缩
KEEP_RECENT_TOOL_RESULTS = 3#只保留最近 3 个完整工具结果
PERSIST_THRESHOLD = 30000#单个工具输出超过 30000 字符时保存到磁盘
PREVIEW_CHARS = 2000#长输出只保留前 2000 个字符作为预览
TRANSCRIPT_DIR = WORKDIR / ".transcripts"#保存完整对话转录的目录
TOOL_RESULTS_DIR = WORKDIR / ".task_outputs" / "tool-results"#保存大型工具输出的目录
```

## 3、定义记录上下文压缩状态的数据类

- **`has_compacted`**：这一轮之前是否已经做过完整压缩
- **`last_summary`**：最近一次压缩得到的摘要
- **`recent_files`**：最近碰过哪些文件，压缩后方便继续追踪

```python
@dataclass
class CompactState:
    has_compacted: bool = False
    last_summary: str = ""
    recent_files: list[str] = field(default_factory=list)
```

```python
def track_recent_file(state: CompactState, path: str) -> None:
		#如果该文件已经在 recent_files 中，先删除旧记录
    if path in state.recent_files:
        state.recent_files.remove(path)
    #再把它追加到列表末尾；
    state.recent_files.append(path)
    #如果超过 5 个文件，只保留最后 5 个。
    if len(state.recent_files) > 5:
        state.recent_files[:] = state.recent_files[-5:]
```

## 6、大输出持久化函数

```python
def persist_large_output(tool_use_id: str, output: str) -> str:
		#如果工具输出长度不超过 PERSIST_THRESHOLD，直接返回完整输出。
    if len(output) <= PERSIST_THRESHOLD:
        return output
    #创建 .task_outputs/tool-results/ 目录；
    TOOL_RESULTS_DIR.mkdir(parents=True, exist_ok=True)
    #使用工具调用 ID 作为文件名,把完整输出保存为 .txt 文件；
    stored_path = TOOL_RESULTS_DIR / f"{tool_use_id}.txt"
    if not stored_path.exists():
        stored_path.write_text(output)
    #标记中包含文件路径和前 PREVIEW_CHARS 个字符的预览。
    preview = output[:PREVIEW_CHARS]
    rel_path = stored_path.relative_to(WORKDIR)
    return (
        "<persisted-output>\n"
        f"Full output saved to: {rel_path}\n"
        "Preview:\n"
        f"{preview}\n"
        "</persisted-output>"
    )
```

## 7、工具结果收集函数

这个函数从历史消息中找出所有工具执行结果,也就是工具结果所在的消息位置、块位置和块本身

```python
def collect_tool_result_blocks(messages: list) -> list[tuple[int, int, dict]]:
    blocks = []
    for message_index, message in enumerate(messages):
        content = message.get("content")
        if message.get("role") != "user" or not isinstance(content, list):
            continue
        for block_index, block in enumerate(content):
            if isinstance(block, dict) and block.get("type") == "tool_result":
                blocks.append((message_index, block_index, block))
    return blocks
```

## 8、工具压缩函数

```python
def micro_compact(messages: list) -> list:
		#先收集所有工具结果
    tool_results = collect_tool_result_blocks(messages)
    #如果工具结果数量不超过 KEEP_RECENT_TOOL_RESULTS，不处理；
    if len(tool_results) <= KEEP_RECENT_TOOL_RESULTS:
        return messages
    #如果超过，则保留最近 3 个；
    for _, _, block in tool_results[:-KEEP_RECENT_TOOL_RESULTS]:
        content = block.get("content", "")
        if not isinstance(content, str) or len(content) <= 120:
            continue
        #更早的工具结果如果内容超过 120 字符，就替换成占位符
        block["content"] = "[Earlier tool result compacted. Re-run the tool if you need full detail.]"
    return messages
```

## 9、写入完整转录函数

这个函数把完整对话历史保存到 `.transcripts/` 目录下。

当代码准备压缩对话时，先把完整历史保存起来，避免原始信息彻底丢失。

```python
def write_transcript(messages: list) -> Path:
    TRANSCRIPT_DIR.mkdir(parents=True, exist_ok=True)
    path = TRANSCRIPT_DIR / f"transcript_{int(time.time())}.jsonl"
    with path.open("w") as handle:
        for message in messages:
            handle.write(json.dumps(message, default=str) + "\n")
    return path
```

## 10、历史总结函数

这个函数调用模型对历史对话进行总结。

总结提示词要求保留五类信息

1. 当前目标
2. 重要发现和决策
3. 读过或改过的文件
4. 剩余工作
5. 用户约束和偏好

```python
def summarize_history(messages: list) -> str:
    conversation = json.dumps(messages, default=str)[:80000]
    prompt = (
        "Summarize this coding-agent conversation so work can continue.\n"
        "Preserve:\n"
        "1. The current goal\n"
        "2. Important findings and decisions\n"
        "3. Files read or changed\n"
        "4. Remaining work\n"
        "5. User constraints and preferences\n"
        "Be compact but concrete.\n\n"
        f"{conversation}"
    )
    response = client.messages.create(
        model=MODEL,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=2000,
    )
    return response.content[0].text.strip()
```

## 11、完整压缩函数

这个函数负责执行完整的上下文压缩。

```python
def compact_history(messages: list, state: CompactState, focus: str | None = None) -> list:
    #保存完整对话转录
    transcript_path = write_transcript(messages)
    print(f"[transcript saved: {transcript_path}]")
    #调用 summarize_history 生成摘要
    summary = summarize_history(messages)
    #如果有 focus，把重点保留内容追加到摘要；
    if focus:
        summary += f"\n\nFocus to preserve next: {focus}"
    #如果有最近文件列表，也追加到摘要；
    if state.recent_files:
        recent_lines = "\n".join(f"- {path}" for path in state.recent_files)
        summary += f"\n\nRecent files to reopen if needed:\n{recent_lines}"
    state.has_compacted = True
    state.last_summary = summary
    return [{
        "role": "user",
        "content": (
            "This conversation was compacted so the agent can continue working.\n\n"
            f"{summary}"
        ),
    }]
```

```python
def execute_tool(block, state: CompactState) -> str:
    if block.name == "bash":
        return run_bash(block.input["command"], block.id)
    if block.name == "read_file":
        return run_read(block.input["path"], block.id, state, block.input.get("limit"))
    if block.name == "write_file":
        return run_write(block.input["path"], block.input["content"])
    if block.name == "edit_file":
        return run_edit(block.input["path"], block.input["old_text"], block.input["new_text"])
    if block.name == "compact":
        return "Compacting conversation..."
    return f"Unknown tool: {block.name}"
```

## 13、Agent结合上下文压缩

```python
def agent_loop(messages: list, state: CompactState) -> None:
    while True:
		    #每轮调用模型前，先压缩较早的工具结果。
        messages[:] = micro_compact(messages)
        #如果上下文太长，就触发自动 compact
        if estimate_context_size(messages) > CONTEXT_LIMIT:
            print("[auto compact]")
            messages[:] = compact_history(messages, state)
        response = client.messages.create(
            model=MODEL,
            system=SYSTEM,
            messages=messages,
            tools=TOOLS,
            max_tokens=8000,
        )
        messages.append({"role": "assistant", "content": response.content})
        if response.stop_reason != "tool_use":
            return
        results = []
        manual_compact = False
        compact_focus = None
        for block in response.content:
            if block.type != "tool_use":
                continue
            output = execute_tool(block, state)
            if block.name == "compact":
		            #启动压缩的开关，后续会根据这个结果判断是否是要启用压缩
                manual_compact = True
                compact_focus = (block.input or {}).get("focus")
            print(f"> {block.name}: {str(output)[:200]}")
            results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": str(output),
            })
        messages.append({"role": "user", "content": results})
        if manual_compact:
            print("[manual compact]")
            messages[:] = compact_history(messages, state, focus=compact_focus)
```
